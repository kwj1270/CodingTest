```c++
#include <iostream>
#include <algorithm>
using namespace std;

int d[10001][3];
int p[10001][3];

int main(void) { 
	
	int n;
	cin >> n;
	cin.ignore();

	for(int i=1;i <=n; i++){
		for(int j=0; j < 3;j++){
			cin >> p[i][j];
	
		}
	}
	for(int i=1; i <=n; i++){
		d[i][0] = min(d[i-1][1],d[i-1][2])+p[i][0];
		d[i][1] = min(d[i-1][0],d[i-1][2])+p[i][1];
		d[i][2] = min(d[i-1][0],d[i-1][1])+p[i][2];
	}
	cout << min(min(d[n][0], d[n][1]), d[n][2]) << "\n";

	return 0;
}
```
