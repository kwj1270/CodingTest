#include <iostream>
#include <vector>

using namespace std;

int main(){
    int money = 15000;
    // vector<int> result(9);
    int result[9] = {0,0,0,0,0,0,0,0,0};
    int sort_money[9] = {50000, 10000, 5000, 1000, 500, 100, 50, 10, 1};
    // 1차 방법
    for(int i=0; i < 9; i++){
        int count = money/sort_money[i];
        if(count == 0) continue;
        result[i] = count;
        money = money%sort_money[i];
    }
    
    for(int i=0; i < 9; i++){
        cout << result[i] << " ";
    }
    
    return 0;
}

#include <iostream>
#include <vector>
#include <queue>
#include <map>

using namespace std;

int main(){
    string word = "I love you";
    string answer = "";
    // 65 부터 알파벳  ~ 90
    map<char, char> m;
    queue<char> q;
    for(int i=0; i <= 25; i++){
        m.insert(make_pair(i+65,90-i));
    }
    for(int i=0; i <= 25; i++){
        m.insert(make_pair(i+97,122-i));
    }
    
    for(int i=0; i < word.size(); i++){
        if(word[i] == ' ') answer += ' ';
        else answer += m[word[i]];
    }
    cout << answer << "\n";
    
    return 0;
}
