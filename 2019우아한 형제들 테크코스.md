# 4. 포비와 크롱 

```c++
#include <iostream>
#include <vector>
#include <queue>
#include <map>
#include <string>

using namespace std;

int sum (int a){
    string str = to_string(a);
    int value = 0;
    for(int i=0; i < str.size(); i++ ){
        value += str[i]-'0';
    }
    return value;
}

int mul (int a){
    string str = to_string(a);
    int value = 1;
    for(int i=0; i < str.size(); i++ ){
        value *= str[i]-'0';
    }
    return value;
}

int main(){
    int pobi [2] = {131, 132};
    int crong [2] = {211, 102};
    
    if(pobi[0] != pobi[1]-1 || crong[0] != crong[1]-1){ cout << -1; return 0;}
    
    int pobiSum =  max(sum(pobi[0]), sum(pobi[1]));
    int crongSum =  max(sum(crong[0]), sum(crong[1]));
    
    int pobiMul =  max(mul(pobi[0]), mul(pobi[1]));
    int crongMul =  max(mul(crong[0]), mul(crong[1]));
    
    int pobiBest = max(pobiSum, pobiMul);
    int crongBest = max(crongSum, crongMul);
    
    if(pobiBest == crongBest) {cout << 0; return 0;}
    int result = pobiBest > crongBest ? 1 : 2;
    
    cout << result << "\n";
    return 0;
}
```

# 티켓팅

```c++
#include <iostream>
#include <vector>
#include <queue>
#include <map>
#include <string>
#include <tuple>
#include <sstream>

using namespace std;

int main(){
    int totalTicek = 2000;
    vector<string> answer;
    vector<string> logos = {
        "woni request 09:12:29",
        "brown request 09:23:11",
        "brown leave 09:23:44",
        "jason leave 09:34:51",
        "jun request 09:33:56",
        "cu request 09:34:02"
    };
    vector<tuple<long long, string, string>> t;// 시간, 이름, 커맨드;
    map<string, long long> m;// 이름, 시간;

    // 55분 시작 -> 4시간 7문제니까 25분 안에 풀어보자
    for(int i=0; i < logos.size(); i++){
        stringstream ss(logos[i]);
        string name;
        string command;
        string time;
        
        ss >> name;
        ss >> command;
        ss >> time;
        
        long long totalTime = 0;
        totalTime += (time[0]-'0'+time[1]-'0') * 60*60;
        totalTime += (time[3]-'0'+time[4]-'0') * 60;
        totalTime += (time[6]-'0'+time[7]-'0');
        
        t.push_back(make_tuple(totalTime, name, command));
    }
    sort(t.begin(), t.end());
 
    for(int i=0; i < t.size(); i++){
        cout << get<0>(t[i]) << " ";
        cout << get<1>(t[i]) << " ";
        cout << get<2>(t[i]) << "\n";
    }
   
    
    int useTime = 0;
    string useUser = "";
    for(int i=0; i < logos.size(); i++){
        long long totalTime = get<0>(t[i]);
        string name = get<1>(t[i]);
        string command = get<2>(t[i]);
        
        if(command == "request"){
            if(useTime+60 < totalTime){
                if(useUser != "" && m[useUser]+60 < totalTime){
                    answer.push_back(useUser);
                    m.erase(useUser);
                }
                useTime = totalTime;
                useUser = name;
                m[name] = totalTime;
            }else {
                if(useUser == name){
                    useTime = totalTime;
                    m[name] = totalTime;
                } else continue;
            }
        }
        else if(command == "leave"){
            if(totalTime - m[name] > 60){
                answer.push_back(name);
            }
            m.erase(name);
            useUser = "";
        }
    }
    
    for(int i=0; i < answer.size(); i++){
        cout << answer[i] << "\n";
    }
    
    return 0;
}
```
