# 4. 포비와 크롱 

```c++
#include <iostream>
#include <vector>
#include <queue>
#include <map>
#include <string>

using namespace std;

int sum (int a){
    string str = to_string(a);
    int value = 0;
    for(int i=0; i < str.size(); i++ ){
        value += str[i]-'0';
    }
    return value;
}

int mul (int a){
    string str = to_string(a);
    int value = 1;
    for(int i=0; i < str.size(); i++ ){
        value *= str[i]-'0';
    }
    return value;
}

int main(){
    int pobi [2] = {131, 132};
    int crong [2] = {211, 102};
    
    if(pobi[0] != pobi[1]-1 || crong[0] != crong[1]-1){ cout << -1; return 0;}
    
    int pobiSum =  max(sum(pobi[0]), sum(pobi[1]));
    int crongSum =  max(sum(crong[0]), sum(crong[1]));
    
    int pobiMul =  max(mul(pobi[0]), mul(pobi[1]));
    int crongMul =  max(mul(crong[0]), mul(crong[1]));
    
    int pobiBest = max(pobiSum, pobiMul);
    int crongBest = max(crongSum, crongMul);
    
    if(pobiBest == crongBest) {cout << 0; return 0;}
    int result = pobiBest > crongBest ? 1 : 2;
    
    cout << result << "\n";
    return 0;
}
```

```c++
#include <iostream>
#include <vector>
#include <queue>
#include <map>
#include <string>

using namespace std;

int getCount(int a){
    string num = to_string(a);
    int count = 0;
    for(int i=0; i < num.size(); i++){
        if(num[i] == '3' || num[i] == '6' || num[i] == '9') count++;
    }
    return count;
}

int main(){
    int number = 33;
    int result = 0;
    for(int i=3; i <= number; i++){
        result += getCount(i);
    }
    cout << result << "\n";
    return 0;
}
```
