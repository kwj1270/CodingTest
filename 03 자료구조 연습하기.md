#include <iostream> 
#include <string> 
#include <stack> 
using namespace std; 


void print(stack<char> &s){
	while(!s.empty()){
		cout << s.top();
		s.pop();
	}
}

int main(void) { 
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    
	string str;
	getline(cin, str);
	bool tag = false;
	stack<char> s;
	for( char ch : str){
		if(ch == '<'){
			print(s); // 이전에 저장된 내용들 출력하라는 뜻이구나!
			tag = true;
			cout << ch; // < 출력 
		} else if( ch == '>'){
			tag = false;
			cout << ch;// > 출력
		} else if (tag){
			cout << ch;
		} else {
			if( ch ==' '){
				print(s);
				cout << ch; // 공백 출력
			} else {
				s.push(ch);
			}
		}
	}
    return 0; 
}
