# 기본적인 올바른 괄호 확인 방법 
```c++
#include <iostream>
#include <vector>
#include <algorithm>
#include <stack>
using namespace std;

int main() {
    int t; cin >> t;
    while (t--) {
        string str; cin >> str;
        stack<char> s;
        bool ok = true;
        for(char ch : str){
            if(ch == '(') s.push(ch);
            else if(!s.empty() && ch == ')') s.pop();
            else if(s.empty() && ch == ')') {ok =false; break;}
        }
        if(!s.empty() || ok == false) {cout << "NO\n"; continue;}
        else cout << "YES\n";
    }

    return 0;
}
```
