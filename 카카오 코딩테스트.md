```c++
    #include <bits/stdc++.h>
    #define INF 987654321
    using namespace std;
    int MAP[25][25][5];  // 진행중인 방향
    bool visited[25][25];
    int dy[4] = {-1, 1, 0, 0};  // 상하좌우
    int dx[4] = {0, 0, -1, 1};
    //직선 도로 하나를 만들 때는 100원이 소요되며, 코너를 하나 만들 때는 500

    int solution(vector<vector<int>> board) {
        int answer = INF;
        int n = board.size();
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                for (int k = 0; k < 4; k++) {
                    MAP[i][j][k] = INF; // 4방향 전부 최대 값을 넣는다.
                }
            }
        }
        for (int k = 0; k < 4; k++) {
            MAP[0][0][k] = 0; // 4방향 전부 시작점부터 0 초기화
        }
        queue<pair<pair<int, int>, int>> q; //3중 큐
        q.push({{0, 0}, 0}); // 3중 큐 푸쉬
        while (!q.empty()) {
            int y = q.front().first.first; //세로
            int x = q.front().first.second; // 가로
            int dir = q.front().second; // 이전 방향을 의마하나보다
            q.pop(); // 큐를 하나 빼고
            visited[y][x] = true; // 방문했다 표시
            for (int k = 0; k < 4; k++) {
                int ny = y + dy[k]; // 다음 세로
                int nx = x + dx[k]; // 다음 가로
                int ndir = k; // 현재 이동될 방향
                if (0 > ny || ny > n - 1 || 0 > nx || nx > n - 1) continue;
                if (board[ny][nx] == 1) continue;  // 벽은 진행 불가
                if (dir == ndir) {  // 방향이 같다면 그대로 전진
                    if (MAP[ny][nx][ndir] > MAP[y][x][dir] + 100) { // 만약 값이 크다면
                        MAP[ny][nx][ndir] = MAP[y][x][dir] + 100; // 값 넣어주고
                        q.push({{ny, nx}, ndir}); // 큐
                    }
                } else {  // 방향이 바뀌었다면
                    if (MAP[ny][nx][ndir] > MAP[y][x][dir] + 600) { // 600보다 클 경우
                        MAP[ny][nx][ndir] = MAP[y][x][dir] + 600; // 600을 넣어준다.
                        q.push({{ny, nx}, ndir});
                    }
                }
            }
        }
        for (int i = 0; i < 4; i++) {
            answer = min(answer, MAP[n - 1][n - 1][i]); // 4방향 중 가장 큰 것
        }
        return answer-500;
    }
```
