			answer += 'L';
			leftHand[left_y][left_x] = 0;

			right_y = row;

				for(int j=0; j < 3;j++){
					if(leftHand[i][j] == 1){
						leftDistance = abs(i-row) + abs(j-column);

					}

			rightHand[right_y][right_x] = 0;
			right_x = column;

				}

			answer += 'L';
			leftHand[left_y][left_x] = 0;
			left_x = column;
			left_y = row;
			leftHand[row][column] = 1;
				}
