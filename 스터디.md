#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int n, k;
vector<int> a;
int d[101];
int main(int argc, const char * argv[]) {
    // 초기화 //
    cin >> n >> k;
    a.resize(n);
    for(int i=0; i < a.size(); i++){
        cin >> a[i];
    }
    sort(a.begin(),a.end());
    // dp 계산 //
    
    d[0] = 1;

    for(int i=0; i < a.size(); i++){ // 저장된 값을 하나씩 뺀다.
        for(int j=1; j <= k; j++){ // 1 ~ 원하는 값까지 도는데
            if(j-a[i] >= 0){ // 현재 반복된 값 - 안에 들어있는 값이 0 이상이라면
                d[j] += d[j-a[i]]; // 현재 반복된 값 == 
            }
        }
    }
    
    cout << d[k] << "\n";
    
    return 0;
}

#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int left_arr[2001];
int right_arr[2001];
int d[2001][2001];

int n;

int cardGame(int left_index, int right_index){
    if(left_index >= n || right_index >= n){
        return 0;
    }
    
    int result = d[left_index][right_index];
    
    result = left_arr[left_index] > right_arr[right_index] ?
    max(result, cardGame(left_index, right_index+1)+right_arr[right_index])
    : max(cardGame(left_index+1, right_index+1), cardGame(left_index+1, right_index));
    
    return result;
}

int main(void) {
    ios_base::sync_with_stdio(0);
    cin.tie(0);

    cin >> n;
    
    for(int i=0; i < n; i++){
        cin >> left_arr[i];
    }
    
    for(int i=0; i < n; i++){
        cin >> right_arr[i];
    }

    cout << cardGame(0, 0) << "\n";
    
}

#include <iostream>
#include <stack>
#include <string>
using namespace std;

struct GoStack {
    int data[10000];
    int size;

    GoStack() {
        size = 0;
    }
    void num(int num) {
        data[size] = num;
        size += 1;
    }
    bool empty() {
        if (size == 0) {
            return true;
        } else {
            return false;
        }
    }
    int pop() {
        if (empty()) {
            return -1;
        } else {
            size -= 1;
            return data[size];
        }
    }
    int top() {
        if (empty()) {
            return -1;
        } else {
            return data[size-1];
        }
    }
};

int main() {
    int n;
    cin >> n;

    GoStack s;

    while (n--) {
        string cmd;
        cin >> cmd;
        
        if (cmd == "NUM") {
            int num;
            cin >> num;
            s.num(num);
        } else if (cmd == "INV") {
            if(s.empty()) cout << "ERROR" << "\n";
            else {
                int temp = -1 * s.top();
                s.pop();
                s.num(temp);
            }
        } else if (cmd == "DUP") {
            if(s.empty()) cout << "ERROR" << "\n";
            else s.num(s.top());
        } else if (cmd == "POP") {
            cout << (s.empty() ? -1 : s.top()) << '\n';
            if (!s.empty()) {
                s.pop();
            }
        } else if (cmd == "SWP") {
            if(s.empty()) cout << "ERROR" << "\n";
            else{
                int x1 = s.top(); s.pop();
                int x2 = s.top(); s.pop();
                s.num(x1);
                s.num(x2);
            }
        } else if (cmd == "ADD") {

                s.num(x1+x2);

                int x1 = s.top(); s.pop();

                s.num(x2-x1);
            }

            if(s.empty()) cout << "ERROR" << "\n";
            else{
                int x1 = s.top(); s.pop();
                int x2 = s.top(); s.pop();

#include <iostream>
#include <stack>
#include <queue>
#include <string>
#include <cmath>

using namespace std;

struct GoStack {
    long long data[1000];
    int size;
    
    // 생성자
    GoStack() {
        size = 0;
    }
    
    // num x
    void num(long long num) {
        data[size] = num;
        size += 1;
    }
    
    // pop
    long long pop() {
        if (empty()) {
            return -1;
        } else {
            size -= 1;
            return data[size];
        }
    }
    
    long long top() {
        if (empty()) {
            return -1;
        } else {
            return data[size-1];
        }
    }
    
    bool inv(){
        if (empty()) {
            return false;
        } else {
            data[size-1] *= -1;
            return true;
        }
    }
    
    bool dup(){
        if (empty()) {
            return false;
        } else {
            data[size] = data[size-1];
            return true;
        }
    }
    
    bool twoValue(string &cmd){
        GoStack s2;
        bool ok = this -> getSize() > 2 ? true : false;
        if(ok == false) return false;
        
        long long x1 = this -> pop();
        long long x2 = this -> pop();
        
        if(cmd == "SWP"){
            this -> num(x1);
            this -> num(x2);
        }
        if(cmd == "ADD"){
            while(this -> empty()){
                s2.num(x2+x1);
                x2 = this -> pop();
            }
            while(s2.empty()){
                this -> num(s2.pop());
            }
        }
        if(cmd == "SUB"){
            this -> num(x2-x1);
        }
        if(cmd == "MUL"){
            this -> num(x2 * x1);
        }
        if(cmd == "DIV"){
            if(x1 == 0) ok = false;
            else if( (x1 < 0 && x2 > 0) || (x1 > 0 && x2 < 0)){
                this -> num(abs(x2)/abs(x1) * -1);
            } else {
                this -> num(abs(x2)/abs(x1));
            }
        }
        if(cmd == "MOD"){
            if(x1 == 0) ok = false;
            else if(x2 < 0){
                this -> num(abs(x2 % x1) * -1);
            } else {
                this -> num(abs(x2)%abs(x1));
            }
        }
        
        return ok;
    }
    
    bool empty() {
        if (size == 0) {
            return true;
        } else {
            return false;
        }
    }
    
    int getSize(){
        return size;
    }
    
};

int main() {
    GoStack s;
    queue<string> cmd_q;
    queue<long long> temp_q;

    while (1) {
        // 명령어 입력 반복 문
        while(1){
            string cmd;
            cin >> cmd;
            if(cmd == "QUIT") cout << "\n"; return 0;
            else if (cmd == "END") break;
            else if (cmd == "NUM") {int num; cin >> num; temp_q.push(num);}
            else if (cmd == "INV" || cmd == "DUP" || cmd == "POP" || cmd == "SWP" ||
                cmd == "ADD" || cmd == "SUB" || cmd == "MUL" || cmd == "DIV" ||
                cmd == "MOD") {
                cmd_q.push(cmd);
            }
        }
        // 값 넣기
        int v;
        cin >> v;
        while(v--){
            long long temp;
            cin >> temp;
            s.num(temp);
        }
        while(temp_q.empty()){
            s.num(temp_q.front()); temp_q.pop();
        }
        
        // 명령어 실행문
        
        while(cmd_q.empty()){
            string cmd = cmd_q.front(); cmd_q.pop();
            if(cmd == "INV") s.inv();
            if(cmd == "DUP") s.dup();
            if(cmd == "POP") s.pop();
            if(cmd == "SWP" || cmd == "ADD" || cmd == "SUB" ||
               cmd == "MUL" || cmd == "DIV" || cmd == "MOD") s.twoValue(cmd);
        }
        
        if(s.getSize() != 1){
            cout << "ERROR" << "\n";
        }
        cout << "\n";
    }
    return 0;
}
