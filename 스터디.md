```c++
#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int n, k;
vector<int> a;
int d[101];
int main(int argc, const char * argv[]) {
    // 초기화 //
    cin >> n >> k;
    a.resize(n);
    for(int i=0; i < a.size(); i++){
        cin >> a[i];
    }
    sort(a.begin(),a.end());
    // dp 계산 //
    
    d[0] = 1;

    for(int i=0; i < a.size(); i++){ // 저장된 값을 하나씩 뺀다.
        for(int j=1; j <= k; j++){ // 1 ~ 원하는 값까지 도는데
            if(j-a[i] >= 0){ // 현재 반복된 값 - 안에 들어있는 값이 0 이상이라면
                d[j] += d[j-a[i]]; // 현재 반복된 값 == 
            }
        }
    }
    
    cout << d[k] << "\n";
    
    return 0;
}
```

```c++
#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int left_arr[2001];
int right_arr[2001];
int d[2001][2001];

int n;

int cardGame(int left_index, int right_index){
    if(left_index >= n || right_index >= n){
        return 0;
    }
    
    int result = d[left_index][right_index];
    
    result = left_arr[left_index] > right_arr[right_index] ?
    max(result, cardGame(left_index, right_index+1)+right_arr[right_index])
    : max(cardGame(left_index+1, right_index+1), cardGame(left_index+1, right_index));
    
    return result;
}

int main(void) {
    ios_base::sync_with_stdio(0);
    cin.tie(0);

    cin >> n;
    
    for(int i=0; i < n; i++){
        cin >> left_arr[i];
    }
    
    for(int i=0; i < n; i++){
        cin >> right_arr[i];
    }

    cout << cardGame(0, 0) << "\n";
    
}
```

```c++
#include <iostream>
#include <stack>
#include <string>
using namespace std;

struct GoStack {
    int data[10000];
    int size;

    GoStack() {
        size = 0;
    }
    void num(int num) {
        data[size] = num;
        size += 1;
    }
    bool empty() {
        if (size == 0) {
            return true;
        } else {
            return false;
        }
    }
    int pop() {
        if (empty()) {
            return -1;
        } else {
            size -= 1;
            return data[size];
        }
    }
    int top() {
        if (empty()) {
            return -1;
        } else {
            return data[size-1];
        }
    }
};

int main() {
    int n;
    cin >> n;

    GoStack s;

    while (n--) {
        string cmd;
        cin >> cmd;
        
        if (cmd == "NUM") {
            int num;
            cin >> num;
            s.num(num);
        } else if (cmd == "INV") {
            if(s.empty()) cout << "ERROR" << "\n";
            else {
                int temp = -1 * s.top();
                s.pop();
                s.num(temp);
            }
        } else if (cmd == "DUP") {
            if(s.empty()) cout << "ERROR" << "\n";
            else s.num(s.top());
        } else if (cmd == "POP") {
            cout << (s.empty() ? -1 : s.top()) << '\n';
            if (!s.empty()) {
                s.pop();
            }
        } else if (cmd == "SWP") {
            if(s.empty()) cout << "ERROR" << "\n";
            else{
                int x1 = s.top(); s.pop();
                int x2 = s.top(); s.pop();
                s.num(x1);
                s.num(x2);
            }
        } else if (cmd == "ADD") {
            if(s.empty()) cout << "ERROR" << "\n";
            else{
                int x1 = s.top(); s.pop();
                int x2 = s.top(); s.pop();
                s.num(x1+x2);
            }
        } else if (cmd == "SUB") {
            if(s.empty()) cout << "ERROR" << "\n";
            else{
                int x1 = s.top(); s.pop();
                int x2 = s.top(); s.pop();
                s.num(x2-x1);
            }
        } else if (cmd == "MUL") {
            if(s.empty()) cout << "ERROR" << "\n";
            else{
                int x1 = s.top(); s.pop();
                int x2 = s.top(); s.pop();
                s.num(x1*x2);
            }
        } else if (cmd == "DIV") {
            if(s.empty()) cout << "ERROR" << "\n";
            else{
                int x1 = s.top(); s.pop();
                int x2 = s.top(); s.pop();
                s.num(x2/x1);
            }
        } else if (cmd == "MOD") {
            if(s.empty()) cout << "ERROR" << "\n";
            else{
                int x1 = s.top(); s.pop();
                int x2 = s.top(); s.pop();
                s.num(x2/x1);
            }
        }
    }
    return 0;
}
```
