# 1 단어 뒤집기 (스택)  
https://www.acmicpc.net/problem/9093  
```c++
#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
#include <stack>

using namespace std;

int main(void){

	int t;
	cin >> t;
	cin.ignore();

	while(t--) {
		string str;
		getline(cin, str);
		str += "\n";

		stack<char> s;
		
		for(int i=0; i < str.length(); i++){
			if(str[i] == ' ' || str[i] == '\n'){
				while(!s.empty()){
					cout << s.top();
					s.pop();
				}
				cout << str[i];
			} else{
			s.push(str[i]);
			}
		}

		while(!s.empty()){
			cout << s.top();
			s.pop();
		}
	}
	return 0;
}
```
# 2 에디터 
https://www.acmicpc.net/problem/1406
```c++
#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
#include <stack>

using namespace std;

int main(void){

	string str;
	getline(cin,str);

	int t;
	cin >> t;
	cin.ignore();
	
	stack<char> left;
	stack<char> right;
	for(int i=0; i<str.length(); i++){
		left.push(str[i]);
	}
	
	while(t--){
		char command;
		cin >> command;
		if(command == 'P'){
			char word;
			cin >> word;
			left.push(word);
		}
		else if(command == 'L'){
			if(!left.empty()){
				right.push(left.top());
				left.pop();
			}
		}
		else if(command == 'D'){
			if(!right.empty()){
				left.push(right.top());
				right.pop();
			}
		}
		else if(command == 'B'){
			if(!left.empty()){
				left.pop();
			}
		}
	}
	while(!left.empty()){
		right.push(left.top());
		left.pop();
	}
	while(!right.empty()){
		cout << right.top();
		right.pop();
	}
	
	return 0;
}
```
# 3 요세푸스 문제
https://www.acmicpc.net/problem/1158
```c++
#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
#include <stack>
#include <queue>

using namespace std;

int main(void){

	queue<int> q;

	int n, k;
	cin >> n >> k;

	for(int i=1; i<=n; i++){
		q.push(i);
	}

	cout << "< ";
	for(int i=0;i < n-1; i++){ // 왜 n-1이냐면 어차피 마지막 1개는 1개이므로  
		for(int j=0; j < k-1; j++){
			q.push(q.front());
			q.pop(); // k가 3이면 2번 
		}
		cout << q.front() << ", ";
		q.pop();
	}
	cout << ">";

	return 0;
}
```
# 4 단어 뒤집기2
```c++
#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
#include <stack>
#include <queue>
#include <deque>

using namespace std;


void print(deque<char> &d){
	while(!d.empty()){
		cout << d.back();
		d.pop_back();
	}
}

int main(void){
	
	string str;
	getline(cin, str);
	str += "\n";
	deque<char> d;

	/////틀린 부분/////
	bool tag = false;
	///////////////////

	for(int i=0; i < str.length(); i++){
		if(str[i] == '<'){
			print(d);
			tag = true;
			cout << str[i];
		} else if(str[i] == '>'){
			tag = false;
			cout << str[i];
		} else if(tag){
			cout << str[i];
		} else { 
			if(!tag && (str[i] == ' ' || str [i] == '\n')){
				print(d);
				cout << str[i];
			} else {
				d.push_back(str[i]);
			}
		}
	}
	print(d);
	cout << "\n";
	return 0;
}
```
