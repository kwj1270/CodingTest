# 1 단어 뒤집기 (스택)  
https://www.acmicpc.net/problem/9093  
```c++
#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
#include <stack>

using namespace std;

int main(void){

	int t;
	cin >> t;
	cin.ignore();

	while(t--) {
		string str;
		getline(cin, str);
		str += "\n";

		stack<char> s;
		
		for(int i=0; i < str.length(); i++){
			if(str[i] == ' ' || str[i] == '\n'){
				while(!s.empty()){
					cout << s.top();
					s.pop();
				}
				cout << str[i];
			} else{
			s.push(str[i]);
			}
		}

		while(!s.empty()){
			cout << s.top();
			s.pop();
		}
	}
	return 0;
}
```
